Get Started

Let’s load the necessary packages:

library(tidyverse)
library(spData)
library(spDataLarge)
library(sf)
library(stars)
library(terra)

And read in the spatial objects:

dem <- terra::rast(system.file("raster/dem.tif", package = "spDataLarge"))
landsat <- terra::rast(system.file("raster/landsat.tif", package = "spDataLarge"))
srtm <- terra::rast(system.file("raster/srtm.tif", package = "spDataLarge"))
# peru_dem <- geodata::elevation_30s(country = "Peru", path = ".", mask = FALSE)
peru <- stars::read_stars(here::here("course-materials", "data", "week4", "PER_elv.tif"))

Now, to meet our learning objectives, your task:

  1. Plot a histogram and boxplot of the DEM raster of Zion National Park from spDataLarge
  2. Reclassify the DEM raster and compute the mean for the three classes:
  • Low (less than 300)
  • Medium
  • High (greater than 500)
  1. Calculate the Normalized Difference Water Index (NDWI) of a Landsat 8 image of Zion National Park
  • Note: \(NDWI = (green - NIR)/(green + NIR)\)
  1. Find a correlation between NDVI and NDWI
  1. Find distances across all peru_dem cells to it’s nearest coastline
  • Hint: Use terra::distance()
  • Note: terra::distance() will calculate distance for all cells that are NA to the nearest cell that are not NA
  1. Weigh the distance raster with peru_dem and visualize the difference between the raster created using the Euclidean distance (E7) and the raster weighted by elevation - Every 100 altitudinal meters should increase the distance to the coast by 10 km
  1. Change the resolution of the SRTM of the Zion National Park to 0.01 by 0.01 degrees
  • Use all of the method available in the terra package
  • Note: The srtm raster has a resolution of 0.00083 by 0.00083 degrees

Answer Key

Plot a histogram and boxplot of the DEM raster

hist(dem)

Reclassify elevation and find the mean

plot(dem)

rcl <- matrix(c(-Inf, 300, 0, 300, 500, 1, 500, Inf, 2), ncol = 3, byrow = TRUE)

dem_rcl <- terra::classify(dem, rcl = rcl)

levels(dem_rcl) <- tibble::tibble(id = 0:2, cats = c("low", "medium", "high"))

elevation_mean <- terra::zonal(dem, dem_rcl, fun = "mean")
elevation_mean

Calculate the NDWI and correlation with NDVI

ndwi_fun <- function(green, nir){
    (green - nir) / (green + nir)
}

ndwi_rast <- terra::lapp(landsat[[c(2, 4)]], fun = ndwi_fun)
plot(ndwi_rast)

ndvi_fun <- function(nir, red){
  (nir - red) / (nir + red)
}

ndvi_rast <- terra::lapp(landsat[[c(4, 3)]], fun = ndvi_fun)

combine <- c(ndvi_rast, ndwi_rast)

plot(combine)

terra::layerCor(combine, fun = cor)

Find distances across all peru_dem cells

peru_dem <- terra::aggregate(peru_dem, fact = 20)
plot(peru_dem)

water_mask <- is.na(peru_dem)
water_mask[water_mask == 0] <- NA
plot(water_mask)

distance_to_coast <- terra::distance(water_mask)
distance_to_coast_km <- distance_to_coast / 1000

plot(distance_to_coast_km, main = "Distance to the coast (km)")

distance_to_coast_km2 <- distance_to_coast_km + ((peru_dem /100) * 10)
plot(distance_to_coast_km2)

Change the resolution of the SRTM

plot(srtm)

rast_template <- terra::rast(terra::ext(srtm), res = 0.01)

srtm_resampl1 <- terra::resample(srtm, y = rast_template, method = "bilinear")
srtm_resampl2 <- terra::resample(srtm, y = rast_template, method = "near")
srtm_resampl3 <- terra::resample(srtm, y = rast_template, method = "cubic")
srtm_resampl4 <- terra::resample(srtm, y = rast_template, method = "cubicspline")
srtm_resampl5 <- terra::resample(srtm, y = rast_template, method = "lanczos")

srtm_resampl_all <- c(srtm_resampl1, srtm_resampl2, srtm_resampl3, srtm_resampl4, srtm_resampl5)
plot(srtm_resampl_all)
